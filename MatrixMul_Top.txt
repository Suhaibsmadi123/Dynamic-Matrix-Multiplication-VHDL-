library ieee;
use ieee.std_logic_1164.all;
use ieee.numeric_std.all;
use work.matrix_pkg.all;

entity MatrixMul_Top is
    generic ( R,N,C : integer );
    port (
        clk, rst, start : in std_logic;
        A_flat : in std_logic_vector(R*N*DATA_WIDTH-1 downto 0);
        B_flat : in std_logic_vector(N*C*DATA_WIDTH-1 downto 0);
        final_result : out std_logic_vector(R*C*DATA_WIDTH-1 downto 0);
        done : out std_logic
    );
end entity;

architecture structural of MatrixMul_Top is

    signal A_slv : array_2d_t(0 to R-1,0 to N-1);
    signal B_slv : array_2d_t(0 to N-1,0 to C-1);
    signal A_int : integer_matrix(0 to R-1,0 to N-1);
    signal B_int : integer_matrix(0 to N-1,0 to C-1);
    signal C_int : integer_matrix(0 to R-1,0 to C-1);

begin

    A_RE : entity work.reshape_1d_to_2d
        generic map(R,N)
        port map(A_flat,A_slv);

    B_RE : entity work.reshape_1d_to_2d
        generic map(N,C)
        port map(B_flat,B_slv);

    process(A_slv,B_slv)
    begin
        for i in 0 to R-1 loop
            for k in 0 to N-1 loop
                A_int(i,k) <= to_integer(signed(A_slv(i,k)));
            end loop;
        end loop;

        for k in 0 to N-1 loop
            for j in 0 to C-1 loop
                B_int(k,j) <= to_integer(signed(B_slv(k,j)));
            end loop;
        end loop;
    end process;

    CORE : entity work.MatrixMul
        generic map(R,N,C)
        port map(clk,rst,start,A_int,B_int,C_int,done);

    process(C_int)
        variable idx : integer := 0;
    begin
        idx := 0;
        for i in 0 to R-1 loop
            for j in 0 to C-1 loop
                final_result((idx+1)*DATA_WIDTH-1 downto idx*DATA_WIDTH)
                    <= std_logic_vector(to_signed(C_int(i,j),DATA_WIDTH));
                idx := idx + 1;
            end loop;
        end loop;
    end process;

end architecture;
